2、 带文字输入统计的textarea文本输入框

textarea的最小高度（umh[类型]）可自行定义。

注意，此组件需要配合js使用。js为cms.js文件中的cms.countChar()方法。


0/10 
                        
<div class="ub ub-ver">
    <div class="ub-f1 uba bc-border cms-input uinn-a04">
        <textarea placeholder="带文字输入统计的textarea文本输入框" type="text" class="ub-f1 umh4" onkeyup="cms.countChar(this,'#currentNum','#totalNum',10);" onkeydown="cms.countChar(this,'#currentNum','#totalNum',10);"></textarea>
    </div>
    <div class="ub ub-f1 ub-pe sc-text uinn-a04 ulev08">
        <span id="currentNum">0</span>/<span id="totalNum">10</span>
    </div>
</div>
<script>
//相对应的js
/**
 * 统计当前输入字数在指定的输入区域显示
 * obj 当前输入框对象
 * @param currentNum 统计字数区域ID
 * @param spanId 显示总共可输入字数区域ID
 * @param wordNumber 限制的输入字数， 字数超过限制变红
 */
cms.countChar = function(obj, currentNum, totalNum, wordNumber) {
    var length = $(obj).val().length;
    $(currentNum).html(length);
    if (length > wordNumber) {
        $(currentNum).addClass("sc-text-active");
    } else {
        $(currentNum).removeClass("sc-text-active");
    }
}
</script>

轮播图片

轮播图片使用的是appcan提供的插件，图片展示的比例，是否有文字描述label，自定义属性都可自行定义。更详细的参数请查询appcan官方文档。

注意：使用此控件须引入appcan.slider.js文件。 

                        
<div id="slider" class="slider"></div>
//轮播图片相应的js
var slider = appcan.slider({
    selector : '#slider', //轮播图片显示区域
    hasLabel : true, //是否有标签文字栏*
    aspectRatio : 6/16, ///是否控制纵横比，0为使用控件高度，>0使用纵横比
    auto : 3000, //轮播时间，单位毫秒，不设不自动滚动
    index : 0 //默认选择项
});
slider.set([{
    img : "../images/test1.jpg",
    label : "第1张图片",
    mydata:"可以自定义属性"
}, {
    img : "../images/test2.jpg",
    label : "第2张图片"
}])
slider.on("clickItem", function(index, data) {
    //点击图片返回的索引及数据
    appcan.window.openToast("你点击了第"+index+"张图片，data为json数据："+ JSON.stringify(data),5000,5,0);
})
